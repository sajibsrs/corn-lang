interface calc {
    intern int sum();
}

interface act {
    extern vec2 *add(int val);
}

struct vec2 {
    int x = 1;
    int y = 2;
}

vec2 <- calc, act; // Implements

vec2 : int sum() {
    return x + y;
}

vec2 : vec2 *add(int val) {
    x = x + val;
    y = y + val;

    return this;
}

# Uses
vec2 v = vec2;

# Everything is public by default
int x = v.x; // 1
int y = v.y; // 2

int sum = v.sum(); // 3
