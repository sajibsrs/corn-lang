# Access Modifiers for Fields & Functions:
# [external] = [default], [internal], [restrict]
# -----------------------------------------------------------------------------
# - They have their generic behavior.
# -----------------------------------------------------------------------------

# Access Modifiers for Types:
# [external] = default, [internal]
# -----------------------------------------------------------------------------
# - [struct], [interface] and [type] etc can have this modifier.
# - [external] which is default, means anyone has access to it as they
#   include/import this file/module.
# - [internal] indicates, it's available exclusively inside that file/module.
# -----------------------------------------------------------------------------

# Interface Declaration:
# [interface] [tag] [{}]
# -----------------------------------------------------------------------------
# - Method declarations should be placed inside the body of the interface.
# - Only the method signature and not it's body.
# - Note: 'self' reserved for interface. So it can refer itself.
# -----------------------------------------------------------------------------
interface ivec2 {
    internal int get_x();
    internal int get_y();
}

# Interface Implementation
internal struct vec2 : ivec2 {
    int x = 1;
    int y = 2;
}

# Method Definition
vec2 : int get_x() {
    return this.x;
}

vec2 : int get_y() {
    return this.y;
}

struct location {
    vec2 pos;
}

# Constructor
# -----------------------------------------------------------------------------
# - When simply called like [vec2 pos = vec2()], that calls the constructor.
#   This allocates static memory and return value. Which is same as calling
#   [vec2 pos = vec2]. If there is no constructor, a struct will always provide
#   implicit one. e.g. [location : location () {}].
#
# - When called like [vec2 *pos = new vec2()] dynamic memory gets allocated,
#   and pointer to the memory returned.
#
# - No explicit return value, implicitly returns [this].
# -----------------------------------------------------------------------------
location : location () {}

# Destructor
# -----------------------------------------------------------------------------
# - No access modifiers.
# - No explicit return value.
# -----------------------------------------------------------------------------
location : ~location () {}

location : vec2 *get_pos() {
    return this.pos;
}

# Uses
vec2 v = vec2;

# Everything is public by default
int x = v.x; // 1
int y = v.y; // 2

int x = v.get_y(); // 1
int y = v.get_y(); // 2

location loc = location();
vec2 pos = loc.get_pos();
int px = pos->x;
int py = pos->y;

location *_loc = new location();
