module main;

import "std:math";

contract action {
   float area();
};

internal struct point {
    internal int x = 1;
    internal int y = 2;
};

struct circle : point {
    float radius;
} action;

circle : circle(float radius) {
    this->radius = radius;
}

circle : @circle() {}

circle : float area() {
    return math:pi * radius ^ 2;
}

circle *pcirc = new circle(10.0);

print(pcirc->radius);
printn(pcirc->area());

purge pcirc;
